112 std::pair<bool, int> find_match(
113 const char* seq,
114 size_t position,
115 int obs_mismatches,
116 const std::array<SimpleVariableLibrary, V>& libs,
117 std::array<typename SimpleVariableLibrary::SearchState, V>& states,
118 std::array<int, V>& temp)
119 const {
120 const auto& regions = constant_matcher.template variable_regions<reverse>();
121
122 for (size_t r = 0; r < V; ++r) {
123 auto range = regions[r];
124 auto start = seq + position;
125 std::string current(start + range.first, start + range.second);
126
127 auto& curstate = states[r];
128 libs[r].match(current, curstate, max_mismatches - obs_mismatches);
129 if (curstate.index < 0) {
130 return std::make_pair(false, 0);
131 }
132
133 obs_mismatches += curstate.mismatches;
134 if (obs_mismatches > max_mismatches) {
135 return std::make_pair(false, 0);
136 }
137
138 if constexpr(reverse) {
139 temp[V - r - 1] = curstate.index;
140 } else {
141 temp[r] = curstate.index;
142 }
143 }
144
145 return std::make_pair(true, obs_mismatches);

179 void process_best(State& state, const std::pair<const char*, const char*>& x) const {
180 auto deets = constant_matcher.initialize(x.first, x.second - x.first);
181 bool found = false;
182 int best_mismatches = max_mismatches + 1;
183 std::array<int, V> best_id;
184
185 auto update = [&](std::pair<int, int> match) -> void {
186 if (match.first && match.second <= best_mismatches) {
187 if (match.second == best_mismatches) {
188 if (best_id != state.temp) { // ambiguous.
189 found = false;
190 }
191 } else {
192 // A further optimization at this point would be to narrow
193 // max_mismatches to the current 'best_mismatches'. But
194 // this probably isn't worth it.
195
196 found = true;
197 best_mismatches = match.second;
198 best_id = state.temp;
199 }
200 }
201 };
202
203 while (!deets.finished) {
204 constant_matcher.next(deets);
205
206 if (forward && deets.forward_mismatches <= max_mismatches) {
207 update(forward_match(x.first, deets, state));
208 }
209
210 if (reverse && deets.reverse_mismatches <= max_mismatches) {
211 update(reverse_match(x.first, deets, state));
212 }
213 }
214
215 if (found) {
216 state.collected.push_back(best_id);
217 }
218 }
