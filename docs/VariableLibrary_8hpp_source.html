23template<class Trie>
24void fill_library(
25 const std::vector<const char*>& options,
26 std::unordered_map<std::string, int>& exact,
27 Trie& trie,
28 bool reverse,
29 bool duplicates
30) {
31 size_t len = trie.get_length();
32
33 for (size_t i = 0; i < options.size(); ++i) {
34 auto ptr = options[i];
35
36 std::string current;
37 if (!reverse) {
38 current = std::string(ptr, ptr + len);
39 } else {
40 for (int j = 0; j < len; ++j) {
41 current += reverse_complement(ptr[len - j - 1]);
42 }
43 }
44
45 auto it = exact.find(current);
46 if (exact.find(current) != exact.end()) {
47 if (!duplicates) {
48 throw std::runtime_error("duplicate variable sequence '" + current + "'");
49 }
50 } else {
51 exact[current] = i;
52 }
53
54 // Note that this must be called, even if the sequence is duplicated;
55 // otherwise the trie's internal counter will not be properly incremented.
56 trie.add(current.c_str(), duplicates);
57 }
58 return;
59}

61template<class Indexer, class Updater, class Cache, class Trie, class Result, class Mismatch>
62void matcher_in_the_rye(const std::string& x, const Cache& cache, const Trie& trie, Result& res, const Mismatch& mismatches, const Mismatch& max_mismatches) {
63 // Seeing if it's any of the caches; otherwise searching the trie.
64 auto cit = cache.find(x);
65 if (cit == cache.end()) {
66 auto lit = res.cache.find(x);
67 if (lit != res.cache.end()) {
68 Updater::update(res, lit->second);
69 } else {
70 auto missed = trie.search(x.c_str(), mismatches);
71
72 // The trie search breaks early when it hits the mismatch cap,
73 // but the cap might be different across calls. If we break
74 // early and report a miss, the miss will be cached and
75 // returned in cases where there is a higher cap (and thus
76 // might actually be a hit). As such, we should only store a
77 // miss in the cache when the requested number of mismatches is
78 // equal to the maximum value specified in the constructor.
79 if (Indexer::index(missed) >= 0 || mismatches == max_mismatches) {
80 res.cache[x] = missed;
81 }
82
83 Updater::update(res, missed);
84 }
85 } else {
86 Updater::update(res, cit->second);
87 }
88 return;
89}
