226 auto deets = constant.initialize(seq, len);
227 state.index = -1;
228 bool found = false;

231 auto update = [&](bool rev, int const_mismatches, const typename SimpleVariableLibrary::SearchState& x) -> void {
232 if (x.index < 0) {
233 return;
234 }
235
236 auto total = x.mismatches + const_mismatches;
237 if (total == best) {
238 if (state.index != x.index) { // ambiguous, setting back to a mismatch.
239 found = false;
240 state.index = -1;
241 }
242 } else if (total < best) {
243 found = true;
244 best = total;
245 // A further optimization at this point would be to narrow
246 // max_mismatches to the current 'best'. But this probably

251 state.variable_mismatches = x.mismatches;
252 state.position = deets.position;

255 };
