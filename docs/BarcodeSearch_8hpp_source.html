23template<class Trie>
24void fill_library(
25 const std::vector<const char*>& options,
26 std::unordered_map<std::string, int>& exact,
27 Trie& trie,
28 bool reverse,
29 bool duplicates
30) {
31 size_t len = trie.get_length();
32
33 for (size_t i = 0; i < options.size(); ++i) {
34 auto ptr = options[i];
35
36 std::string current;
37 if (!reverse) {
38 current = std::string(ptr, ptr + len);
39 } else {
40 for (int j = 0; j < len; ++j) {
41 current += reverse_complement(ptr[len - j - 1]);
42 }
43 }
44
45 auto it = exact.find(current);
46 if (exact.find(current) != exact.end()) {
47 if (!duplicates) {
48 throw std::runtime_error("duplicate variable sequence '" + current + "'");
49 }
50 } else {
51 exact[current] = i;
52 }
53
54 // Note that this must be called, even if the sequence is duplicated;
55 // otherwise the trie's internal counter will not be properly incremented.
56 trie.add(current.c_str(), duplicates);
57 }
58 return;
59}

61template<class Methods, class Cache, class Trie, class Result, class Mismatch>
62void matcher_in_the_rye(const std::string& x, const Cache& cache, const Trie& trie, Result& res, const Mismatch& mismatches, const Mismatch& max_mismatches) {
63 // Seeing if it's any of the caches; otherwise searching the trie.
64 auto cit = cache.find(x);
65 if (cit == cache.end()) {
66 auto lit = res.cache.find(x);
67 if (lit != res.cache.end()) {
68 Methods::update(res, lit->second, mismatches);
69
70 } else {
71 auto missed = trie.search(x.c_str(), mismatches);
72
73 // The trie search breaks early when it hits the mismatch cap,
74 // but the cap might be different across calls. If we break
75 // early and report a miss, the miss will be cached and
76 // returned in cases where there is a higher cap (and thus
77 // might actually be a hit). As such, we should only store a
78 // miss in the cache when the requested number of mismatches is
79 // equal to the maximum value specified in the constructor.
80 if (Methods::index(missed) >= 0 || mismatches == max_mismatches) {
81 res.cache[x] = missed;
82 }
83
84 Methods::update(res, missed);
85 }
86 } else {
87 Methods::update(res, cit->second, mismatches);
88 }
89 return;
90}
