225 static int strand_match(const MatchDetails& match, const std::bitset<N>& ref, const std::bitset<N>& mask) {
226 // pop count here is equal to the number of non-ambiguous mismatches *
227 // 2 + number of ambiguous mismatches * 3. This is because
228 // non-ambiguous bases are encoded by 1 set bit per 4 bases (so 2 are
229 // left after a XOR'd mismatch), while ambiguous mismatches are encoded
230 // by all set bits per 4 bases (which means that 3 are left after XOR).
231 int pcount = ((match.state & mask) ^ ref).count();
232
233 // Counting the number of ambiguous bases after masking. Each ambiguous
234 // base is represented by 4 set bits, so we divide by 4 to get the number
235 // of bases; then we multiply by three to remove their contribution. The
236 // difference is then divided by two to get the number of non-ambig mm's.
237 if (!match.bad.empty()) {
238 int acount = (match.ambiguous & mask).count();
239 acount /= 4;
240 return acount + (pcount - acount * 3) / 2;
241 } else {
242 return pcount / 2;
243 }
244 }
